local Library = loadstring(game:HttpGet("https://raw.githubusercontent.com/xHeptc/Kavo-UI-Library/main/source.lua"))()
local Window = Library.CreateLib("Sneezing Simulator Script | by Fxnn Over Heaven#5857", "Ocean")

local Tab = Window:NewTab("Main")
local Section = Tab:NewSection("AutoSneeze")

Section:NewButton("Infinitie Points", "eZ Points", function()
    local a = game:GetService("Players").LocalPlayer.PlayerGui.Client
local code

for i,v in pairs(getreg())do
if typeof(v) == 'function' and getfenv(v).script == a then
local b = debug.getupvalues(v)
for i2,v2 in pairs(b)do
if typeof(v2) == 'number' and string.len(tostring(v2)) == 10 then
code = v2
break
end
end
end
end
game:GetService("RunService").Heartbeat:Connect(function()
game:GetService("ReplicatedStorage").Events.Infect:FireServer(code, true, true)
end)
wait()

game:GetService("RunService").Heartbeat:Connect(function()
game:GetService("ReplicatedStorage").Events.Bonus:FireServer(code, true)
end)
end)

local Tab = Window:NewTab("Player")
local Section = Tab:NewSection("Local Player")

Section:NewSlider("WalkSpeed", "Changes your WalkSpeed", 250, 16, function(s) 
    game.Players.LocalPlayer.Character.Humanoid.WalkSpeed = s
end)

Section:NewSlider("JumpPower", "Changes your JumpPower", 250, 16, function(s) 
    game.Players.LocalPlayer.Character.Humanoid.JumpPower = s
end)
