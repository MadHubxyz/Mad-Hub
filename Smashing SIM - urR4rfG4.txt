local Library = loadstring(game:HttpGet("https://raw.githubusercontent.com/xHeptc/Kavo-UI-Library/main/source.lua"))()
local Window = Library.CreateLib("Smashing Simulator X Script | by Fxnn Over Heaven#5857", "Ocean")
local Tab = Window:NewTab("Main")
local Section = Tab:NewSection("Throw Faster!")
Section:NewButton("FasterThrow", "U can Throw Ur Item faster", function()
    for i,v in pairs(getgc(true)) do
        if type(v) =="table" and rawget(v,"Force") ~=nil then
            v.Force = 5000
            v.ThrowDebounce = 0
        end
    end
        end)

local Tab = Window:NewTab("Player")
local Section = Tab:NewSection("local Player")

Section:NewSlider("WalkSpeed", "Speeds Ur Character", 500, 16, function(s) 
    game.Players.LocalPlayer.Character.Humanoid.WalkSpeed = s
end)

Section:NewSlider("JumpPower", "Jump higher", 500, 16, function(s) 
    game.Players.LocalPlayer.Character.Humanoid.JumpPower = s
end)

local Tab = Window:NewTab("UI Toggles")
local Section = Tab:NewSection("Set ur Ui KeyBinds!")

Section:NewKeybind("UI Toggle KeyBinds", "KeybindSet", Enum.KeyCode.RightAlt, function()
	Library:ToggleUI()
end)
